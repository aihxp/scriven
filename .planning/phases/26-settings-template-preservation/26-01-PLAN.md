         try {

       function mergeSettings(existing, incoming, schema = SETTINGS_SCHEMA) {
         const merged = { ...incoming };
         if (!existing || typeof existing !== 'object') return merged;
         for (const [key, value] of Object.entries(existing)) {
           if (INSTALLER_OWNED_FIELDS.includes(key)) continue; // installer-owned: incoming wins

       }
